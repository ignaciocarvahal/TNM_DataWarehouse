SELECT b.id, 
    b.sii_factura, 
    d.fk_servicio, 
    b.estado, b.total_neto, b.cant_servicios,
    b.total_servicios, b.total_cuadrilla, b.total_sobrepeso, b.total_sobreeestadia,
    b.total_almacenaje, b.total_con_refer, b.total_otros_cobros, b."createdAt", b."updatedAt",
    b.fk_responsable, b.fk_cliente, b.sii_fecha, b.fecha_envio_fact, b.factura_num_cont, b.factura_nave, 
    b.factura_ref_cliente, b.factura_ref_despacho, b.factura_sub_cliente, b.factura_o_c, b.factura_o_c_fecha, 
    b.max_id_interno, b.tipo,
	
	d.cliente_despacho_codigo, 
	d.cliente_despacho_rut,
    d.cliente_despacho_nombre,
	d.nave_id, 
	d.nave_nombre,
	d.contenedor,
    d.age_nombre, 
	d.cont_tamano,
    d.total_viajes_1, 
	d.total_viajes_1_refeer, 
	d.total_viajes_2, 
	d.cuadrilla,
    d.cuadrilla_cantidad, 
	d.sobre_peso,
	d.sobre_estadia_valor,
    d.sobreestadia_horas_libres, 
	d.sobreestadia_costo_hora,
	d.almacenaje_dias,
    d.almacenaje_dias_valor, 
	d.almacenaje_dias_valor_conexion,
    d.almacenaje_dias_conexion_horas, 
	d.total, 
	d.total_cobros_extras,
    d."createdAt" AS detalle_createdAt,
	d."updatedAt" AS detalle_updatedAt,
	d.fk_responsable AS detalle_fk_responsable,
	d.fk_liquidacion, 
	d.fk_propuesta
FROM public.facturacion_liquidaciones_bases AS b
LEFT JOIN public.facturacion_liquidaciones_detalles AS d
ON b.id = d.fk_liquidacion;
